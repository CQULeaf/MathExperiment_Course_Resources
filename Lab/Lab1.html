import numpy as np

# 
def verify_determinant_expansion(A):
    det_A = np.linalg.det(A) # 
    expansion_sum = 0
    for i in range(len(A)):
        minor = np.linalg.det(np.delete(np.delete(A, 0, axis=0), i, axis=1))  # 0i
        cofactor = ((-1)**(i)) * minor # 
        expansion_sum += A[0, i] * cofactor # 
    close = np.isclose(det_A, expansion_sum) # 
    return det_A, expansion_sum, close

# 
def verify_cramers_rule(A, b):
    det_A = np.linalg.det(A) # 
    # 0
    if det_A == 0:
        return "System does not have a unique solution"
    # 
    solutions = []
    for i in range(len(A)):
        Ai = np.copy(A)
        Ai[:, i] = b
        det_Ai = np.linalg.det(Ai)
        x_i = det_Ai / det_A
        solutions.append(x_i)
    return solutions

# 
A = np.array([[2, 1, -1],
              [-3, -1, 2],
              [-2, 1, 2]])

b = np.array([8, -11, -3])

# 
det_A, expansion_sum, close_1 = verify_determinant_expansion(A)

#

    solutions = verify_cramers_rule(A, b)

def is_narcissistic_number(num):
    # 
    num_str = str(num)
    num_length = len(num_str)
    # 
    sum_of_powers = sum(int(digit) ** num_length for digit in num_str)
    return sum_of_powers == num

    # 
    min_diff = pi
    closest_fraction = None

    # 
    for denom in range(1, max_val + 1):
        for numer in range(min_val, max_val + 1):
            fraction = Fraction(numer, denom)
            diff = abs(float(fraction) - pi)

            if diff < min_diff:
                min_diff = diff
                closest_fraction = fraction
                
            #

            if diff < min_diff:
                min_diff = diff
                closest_fraction = fraction

    return closest_fraction

    sum = 0
    for i in range(n + 1):
        sum += np.exp(1) * ((x - 1)**i) / math.factorial(i)
    return sum

def collatz_steps(n):
    steps = 0
    while n != 1:
        n = n / 2 if n % 2 == 0 else 3 * n + 1
        steps += 1
    return steps
